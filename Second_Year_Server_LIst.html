import turtle
 
def callMike(t):
	
		
		t.penup()
		t.goto(0,0)
		t.pendown()
		count = 0
		t.width(10)
		t.color("#000000")
		t.circle(100)
		t.penup()
		t.goto(50,50)
		t.pendown()
		t.color("#0000ff")
		t.goto(-50,50)
		t.goto(-50,-50)
		t.goto(50,-50)
		t.goto(50,50)
		t.color("#00ff00")
		t.goto(0,0)
		t.goto(0,-200)
	
	
		
def main():
	try:
		turtle.TurtleScreen._RUNNING = True
		drawing_area = turtle.Screen()
		drawing_area.setup(width=700, height=700)
		w = turtle.Screen()
		w.clear()
		w.title("one")
		w.bgcolor("#FF0000")
		t = turtle.Turtle()
		callMike(t)
		w.exitonclick()
	finally:
		turtle.Terminator()
		
if __name__ == '__main__':
	main()
